d)

przekształcenie transformaty Fouriera sygnału po odjęciu trendu

 fftResids = fft(z);

 % przekształcenie transformaty Fouriera na sposób dający się wyświetlić w
 % postaci interpretowalej dla człowieka
 L = length(fftResids);
 P2 = abs(fftResids/L);
 P1 = P2(1:(L/2)+1);
 P1(2:end-1) = 2*P1(2:end-1);

 % częstotliwość próbkowania wzięta z pliku dane.csv
 Fs = 900;
 f = Fs*(0:(L/2))/L;
 % wykres
 figure(7);
 plot(f, P1);
 hold on
 plot(72.83, 0.543, "r*");
 text(72.83, 0.543, "Fc ~= 72 Hz");
 title('FFT Sygnalu');
 xlabel('f (Hz)');
 ylabel('|Y(f)|');
